function MDC(numUm : int , numDois : int, resultado : int) : void {
	if (numUm == 0){
		resultado = numDois;
	}
	else{
		var resto:int;
		resto = numUm%numDois;
		MDC(resto,numUm, resultado);
	};
};
main() {
	var intUM :int = 0; var intDois :int = 0;
    var res:int;
    input("%d", &intUM);
    input("%d", &intDois);
    MDC(intUM, intDois, &res);

    output("O MDC eh %d\n", &res);

    return 0;
}
